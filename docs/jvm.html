show_state = (rs) ->
  cf = rs.curr_frame()
  s = ((if x?.ref? then x.ref else x) for x in cf.stack)
  l = ((if x?.ref? then x.ref else x) for x in cf.locals)
  debug "stack: [#{s}], locals: [#{l}]"
